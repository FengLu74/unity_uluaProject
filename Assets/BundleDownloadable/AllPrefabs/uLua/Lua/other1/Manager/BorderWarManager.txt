module("BorderWarManager", package.seeall)

-- 边境战场大任务数据
BorderWarMissionTypeData = {};
-- 边境战场子任务数据
BorderWarMissionData = {};
-- 边境战场其他数据
BorderWarData = {};
-- 宝箱状态(默认为0未领取状态)
BorderWarBoxState = 0;

-- 清理数据
function ClearData()
    BorderWarMissionTypeData = {};
    BorderWarMissionData = {};
    BorderWarData = {};
    BorderWarBoxState = 0;
end

-- 该任务是否需要提示
function IsNeedShowTip(taskID)
    for i,v in pairs(BorderWarGatherTaskID) do
        if taskID == v then 
            return true;
        end
    end
    return false;
end

-- 该任务是否击杀任务
function IsCampKillTask(taskID)
    for i,v in pairs(BorderWarKillTaskID) do
        if taskID == v then 
            return true;
        end
    end
    return false;
end
